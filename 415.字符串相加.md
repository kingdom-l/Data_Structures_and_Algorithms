415.字符串相加

给定两个字符串形式的非负整数 num1 和num2 ，计算它们的和并同样以字符串形式返回。

你不能使用任何內建的用于处理大整数的库（比如 BigInteger）， 也不能直接将输入的字符串转换为整数形式

思路：由于num1的元素存放的是整数的高位到低位，因此，从后往前遍历来实现加法。

```C++
class Solution {
public:
    void reverse(string &str){
        int i = 0, j = str.size() - 1;    
        while(i < j){
            char tem = str[i];
            str[i++] = str[j];
            str[j--] = tem;
        }
    }
    string addStrings(string num1, string num2) {
        string ans;
        int c = 0;
        int i = num1.size() - 1, j = num2.size() - 1;
        int sum = 0, rem = 0;
        for(; i >= 0 && j >= 0; --i, --j){
            sum = num1[i] - '0' + num2[j] - '0' + c;
            rem = sum % 10;
            c = sum / 10;
            ans += rem + '0';
        }
        while(i >= 0){
            sum = num1[i--] - '0' + c;
            rem = sum % 10;// 求余
            ans += rem + '0';
            c = sum / 10;
        }
        while(j >= 0){
            sum = num2[j--] - '0' + c;
            rem = sum % 10;// 求余
            ans += rem + '0';
            c = sum / 10;
        }
        if(c){
            ans += '0' + c; //对应例子‘9’+‘1’
        }
        reverse(ans); //翻转ans
        return ans;
    }
};
```

class Solution {
public:
    void reverse(string &str){
        int i = 0, j = str.size() - 1;    
        while(i < j){
            char tem = str[i];
            str[i++] = str[j];
            str[j--] = tem;
        }
    }
    string addStrings(string num1, string num2) {
        string ans;
        int c = 0;
        int i = num1.size() - 1, j = num2.size() - 1;
        int sum = 0, rem = 0;
        for(; i >= 0 || j >= 0 || c != 0; --i, --j){
        	int x = i >= 0 ? num1[i] - '0' : 0;
        	int y = j >= 0 ? num2[j] - '0' : 0;
            sum = x + y + c;
            rem = sum % 10;
            c = sum / 10;
            ans += rem + '0';
        }
        reverse(ans); //翻转ans
        return ans;
    }
};
