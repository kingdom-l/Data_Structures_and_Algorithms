合并两个有序链表

规则：将两个升序链表合并成一个新的升序链表

比如：

```
输入：l1 = [1,2,4], l2 = [1,3,4]
输出：[1,1,2,3,4,4]
```

思路1：建立一个虚拟节点，然后比较链表l1和链表l2的节点，将比较小的值加入到虚拟节点后面

```C++
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* list1, ListNode* list2) {
        ListNode* vir = new ListNode(-1);
        ListNode* head = vir;
        while(list1 != nullptr || list2 != nullptr){
            if(list1 != nullptr && list2 != nullptr){
                if(list1->val > list2->val){
                    ListNode* list3 = new ListNode(list2->val);
                    head->next = list3;
                    list2 = list2->next;
                }else if((list1->val < list2->val)){
                    ListNode* list3 = new ListNode(list1->val);
                    head->next = list3;
                    list1 = list1->next;
                }else{
                    ListNode* list3 = new ListNode(list1->val);
                    head->next = list3;
                    head = head->next;
                    ListNode* list4 = new ListNode(list1->val);
                    head->next = list4;
                    list1 = list1->next;
                    list2 = list2->next;
                }
                
            }else if(list1 == nullptr){         
                head->next = list2;
                list2 = list2->next;
            }else{         
                head->next = list1;
                list1 = list1->next;
            }
            head = head->next;
        }
        return vir->next;
    }
};
```

思路2：通过递归实现

```C++
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* list1, ListNode* list2) {
        //终止条件
        if(list1 == NULL){
            return list2; 
        }
        if(list2 == NULL){
            return list1;
        }
        if(list1->val > list2->val){
            list2->next = mergeTwoLists(list1, list2->next);
            return list2;
        }
        
        list1->next = mergeTwoLists(list1->next, list2);
        return list1;
    }
};
```
