无重复字符最长子串

```C++
class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        unordered_set<char> occ;
        int n = s.size();
        //定义右边界
        int right = -1, ans = 0;
        for(int i = 0; i < n; ++i){
            //左指针向右移动一个位置
            if(i != 0){
                occ.erase(s[i - 1]);
            }
            //右指针向右移动一个位置，同时判断是否出现重复
            //如果没有重复，则将这个元素加入，并移动右指针指向这个元素
            while(right + 1 < n && !occ.count(s[right + 1])){
                occ.insert(s[right + 1]);
                ++right;
            }
            //更新最长子串的长度
            ans = max(ans, right - i + 1);
        }
        return ans;
    }
};
```
