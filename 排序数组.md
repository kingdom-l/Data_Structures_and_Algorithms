排序数组

给一个数组num，请将该数组升序排列

思路一：快排，选择一个基准值，使得基准值左边的元素小于基准值，右边的元素大于基准值。

```C++
class Solution {
public:
    void quicksort(vector<int>& nums, int l, int r){
        if(l >= r){
            return;
        }
        int ref_index = random(nums, l, r);
        quicksort(nums, l, ref_index - 1);
        quicksort(nums, ref_index + 1, r);
    }
    int random(vector<int>& nums, int l, int r){
        int ran_index = rand() % (r - l + 1) + l;
        swap(nums[ran_index], nums[r]);
        return partition(nums, l, r);
    }
    int partition(vector<int>& nums, int l, int r){
        int ref_val = nums[r];
        int start = l - 1;//记录大于ref_val的下标
        //int end = r;
        for(int i = l; i < r; ++i){
            if(nums[i] < ref_val){
                swap(nums[++start], nums[i]);
            }
        }
        swap(nums[++start], nums[r]);
        return start;
    }
    vector<int> sortArray(vector<int>& nums) {
        srand(time(0));
        quicksort(nums, 0, nums.size() - 1);
        return nums;
    }
};
```
