23.合并K个升序链表

给你一个链表数组，每个链表都已经按升序排列。

请你将所有链表合并到一个升序链表中，返回合并后的链表。

思路1：顺序合并，时间复杂度为O($k^2n$)

```C++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    // 时间复杂度为O(n)，空间复杂度为O(n)
    ListNode* mergeTwoLists(ListNode* head1, ListNode* head2){
        ListNode* ans = new ListNode(-1);
        ListNode* vir = ans;
        while(head1 && head2){
            if(head1->val > head2->val){
                ListNode* temp1 = new ListNode(head2->val);
                ans->next = temp1;
                head2 = head2->next;
            }else{
                ListNode* temp2 = new ListNode(head1->val);
                ans->next = temp2;               
                head1 = head1->next;
            }
            ans = ans->next;
        }
        //while(head1){
        //    ans->next = head1;
        //    head1 = head1->next;
        //    ans = ans->next;
        //}
        //while(head2){
        //    ans->next = head2;
        //    head2 = head2->next;
        //    ans = ans->next;
        //}
        ans->next = head1 ? head1 : head2;
        return vir->next;
    }
    ListNode* mergeKLists(vector<ListNode*>& lists) {
        //问题：由于lists的个数是变化的，因此for循环嵌套也是不定的
        //ListNode* head;
        if(lists.size() == 0){
            return NULL;
        }
        for(int i = 0; i < lists.size() - 1; ++i){
            lists[i+1] = mergeTwoLists(lists[i], lists[i+1]);
        }
        return lists[lists.size() - 1];
    }
};
```

思路2：第一轮合并是将k个链表两两合并，得到$\frac{k}{2}$，每组的时间代价是O(2n)；第二轮合并是将$\frac{k}{2}$合并成$\frac{k}{4}$组链表，每组的时间代价是O(4n)，总的时间代价是O($\sum_{i = 1}^{log k} \frac{k}{2^i} \times 2^in$)，使用栈空间的代价为O($log k$)。

```C++
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* head1, ListNode* head2){
        ListNode* ans = new ListNode(-1);
        ListNode* vir = ans;
        while(head1 && head2){
            if(head1->val > head2->val){
                ListNode* temp1 = new ListNode(head2->val);
                ans->next = temp1;
                head2 = head2->next;
            }else{
                ListNode* temp2 = new ListNode(head1->val);
                ans->next = temp2;               
                head1 = head1->next;
            }
            ans = ans->next;
        }
        ans->next = head1 ? head1 : head2;
        return vir->next;
    }
    ListNode* merge(vector<ListNode*>& list, int l, int r){
        if(l == r){
            return list[l];
        }
        if(l > r){
            return NULL;
        }
        int mid = (l + r) >> 1;
        return mergeTwoLists(merge(list, l, mid), merge(list, mid+1, r));
    }
    ListNode* mergeKLists(vector<ListNode*>& lists) {
        //问题：由于lists的个数是变化的，因此for循环嵌套也是不定的
        //ListNode* head;
        return merge(lists, 0, lists.size() - 1);
    }
};
```

思路3：将lists存放的k个链表的头节点添加进优先队列中，然后重载优先队列中的比较函数，使得最小元素排放在队头，之后每次弹出优先队列中最小节点。优先队列中插入和删除元素的时间复杂度为O($logk$)，而最多有$kn$个节点插入到优先队列中，因此，总的时间复杂度为O($kn \times logk$)。空间复杂度为O(k)。

```C++
class Solution {
public:
    class Status{
    public:
        int val;
        ListNode* node;
        Status(int _val, ListNode* _node) : val(_val), node(_node){}
        bool operator < (const Status& slist1) const {
            return val > slist1.val;
        }
    };
    priority_queue<Status> plist;
    ListNode* mergeKLists(vector<ListNode*>& lists) {
        ListNode head, *tail = &head;
        for(auto lis : lists){
            // 将lists存放的k个链表的头节点添加进优先队列中
            if(lis){
                plist.push(Status(lis->val, lis));
            }
        }
        // 此时优先队列的队头元素为最小节点，也是这k个链表中最小节点（因为链表是升序排列）
        while(!plist.empty()){
            auto lis1 = plist.top(); plist.pop();
            tail->next = lis1.node;
            tail = tail->next;
            if(lis1.node->next){
                plist.push({lis1.node->next->val, lis1.node->next});
            }
        }
        return head.next;
    }
};
```
